				}

				who = none
			}
			break = yes
		}

			limit = {
				NOT = { is_same_species = from }

				}
			}

				who = none

			break = yes
		}

			limit = {
				from = {

				who = from

			break = yes

			}
			set_subject_of = {
				who = from
				subject_type = signatory
			}
			break = yes

			limit = {

				}

			}

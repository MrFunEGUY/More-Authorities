# Term is up
country_event = {
	id = moreauthorities.4
	picture = GFX_evt_alien_propaganda
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		exists = ruler
		ruler = {
			has_leader_flag = one_term
		}
		has_authority = auth_strict_democratic
		has_country_flag = has_term_limits
	}
	
	immediate = {
		hidden_effect = {
			ruler = { save_event_target_as = current_president }
			if = {
				limit = { ruler = { pre_ruler_leader_class = scientist } }
				clone_leader = {
					target = event_target:current_president
					class = scientist
					event_leader = yes
				}
			}
			else_if = {
				limit = { ruler = { pre_ruler_leader_class = general } }
				clone_leader = {
					target = event_target:current_president
					class = general
					event_leader = yes
				}
			}
			else_if = {
				limit = { ruler = { pre_ruler_leader_class = admiral } }
				clone_leader = {
					target = event_target:current_president
					class = admiral
					event_leader = yes
				}
			}
			else = {
				clone_leader = {
					target = event_target:current_president
					class = governor
					event_leader = yes
				}
			}
			ruler = {
				kill_leader = { show_notification = no }
			}
		}
	}
}

	desc = moreauthorities.5.desc
	picture = GFX_evt_throne_room
	is_triggered_only = yes
		
	option = {
		name = moreauthorities.5.a
		allow = {
		custom_tooltip = {
				fail_text = influencefail_text_ma_1
				has_resource = { type = influence amount >= 450 }
			}
		}
		hidden_effect = {
			add_resource = { influence = -450 }
			country_add_ethic = ethic_egalitarian
			change_government = {
				authority = auth_strict_democratic

			}
			set_government_cooldown = 1800
			set_country_flag = has_term_limits
			ruler = { set_leader_flag = one_term }
		}	
	}

	option = {
		name = moreauthorities.5.b
	}
}

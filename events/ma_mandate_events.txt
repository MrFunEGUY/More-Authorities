# Mandate added
country_event = {
	id = mandate.1
	title = mandate.1.title
	desc = mandate.1.desc
	picture = GFX_evt_throne_room
	show_sound = event_default

	is_triggered_only = yes

	# "ruler X elected, check sitch log for their mandate"

	trigger = {
		exists = ruler
		NOT = {
			has_authority = auth_democratic_dictatorship
		}
		ruler = {
			has_mandate = yes
		}
	}

	option = {
		name = OK
	}
}
